Get-ChildItem -Recurse -Filter *.json .\src\page-maps | ForEach-Object {
  try {
    $raw = Get-Content $_.FullName -Raw
    $j = $raw | ConvertFrom-Json

    if (-not $j.pageKey -or -not ($j.pageKey -is [string])) {
      Write-Host "❌ Missing/invalid pageKey:" $_.FullName
    }

    if (-not $j.elements) {
      Write-Host "❌ Missing elements:" $_.FullName
    }
  }
  catch {
    Write-Host "❌ Invalid JSON:" $_.FullName
  }
}
